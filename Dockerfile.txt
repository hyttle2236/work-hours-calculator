# 使用 Python 环境
FROM python:3.11-slim

# 设置工作目录
WORKDIR /app

# 复制依赖列表
COPY requirements.txt .

# 安装依赖 (防止缓存问题)
RUN pip install --no-cache-dir -r requirements.txt

# 复制所有代码
COPY . .

# 设置环境变量，确保 Flet 知道它是 Web 服务
ENV FLET_SERVER_PORT=8080
ENV PORT=8080

# 暴露端口
EXPOSE 8080

# 启动命令
CMD ["python", "main.py"]
